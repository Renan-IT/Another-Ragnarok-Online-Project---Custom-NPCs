//=NPC WARPER ALBERTA=//

alberta,114,65,4	script	Warper#al	4_F_NURSE,{
set @npc$, "[Warper]"; //Nom NPC

	if (alreadytalk ==1){
	mes @npc$;
	mes "Nous préparons votre monture !";
	close;
	}
	else{
		mes @npc$;
		mes "Bonjour "+strcharinfo(0)+",";
		mes "Je peux vous transporter à dos de Peco Peco dans les villes alentours si vous le souhaitez.";
		mes "Les villes accéssibles ici sont : Izlude, Jawaii et Payon.";
		next;
		mes @npc$;
		mes "Dans quelles villes souhaitez-vous aller ?";
		switch(select("Izlude","Jawaii","Payon","Aucune")){
			case 1:
				next;
				mes @npc$;
				mes "très bien je demande tout suite à un écuyer de vous préparer votre Peco Peco !";
				mes "Il sera prêt dans 20s environ,";
				mes "pendant ce temps là, prenez le temps de préparer 	votre équipement !";
				addtimer 20000, strnpcinfo(3)+"::OnWarp";
				addtimer 15000, strnpcinfo(3)+"::OnSec";
				set izlude, 1;
				//warp après 20s
				set alreadytalk, 1;
				close;
			case 2:
				next;
				mes @npc$;
				mes "très bien je demande tout suite à un écuyer de vous préparer votre Peco Peco !";
				mes "Il sera prêt dans 20s environ,";
				mes "pendant ce temps là, prenez le temps de préparer votre équipement !";
				addtimer 20000, strnpcinfo(3)+"::OnWarp";
				addtimer 15000, strnpcinfo(3)+"::OnSec";
				set jawaii, 1;
				//warp après 20s
				set alreadytalk, 1;
				close;
			case 3:
				next;
				mes @npc$;
				mes "très bien je demande tout suite à un écuyer de vous 	préparer votre Peco Peco !";
				mes "Il sera prêt dans 20s environ,";
				mes "pendant ce temps là, prenez le temps de préparer 	votre équipement !";
				addtimer 20000, strnpcinfo(3)+"::OnWarp";
				addtimer 15000, strnpcinfo(3)+"::OnSec";
				set payon, 1;
				//warp après 20s
				set alreadytalk, 1;
				close;
			case 4:
				mes @npc$;
				mes "A bientôt j'espère !";
				close;
		}
	}
	OnWarp:
		dispbottom "Vous allez être warp";
		if (izlude == 1){
			set alreadytalk, 0;
			warp "izlude",128,142;
		}
		else if (jawaii == 1){
			set alreadytalk, 0;
			warp "jawaii",181,161;
		}
		else if (payon == 1){
			set alreadytalk, 0;
			warp "payon",175,102;
		}
	end;

	OnSec:
		dispbottom "Encore 5 secondes avant d'être téléporté, préparez-vous !";
	end;
	
}

//=NPC WARPER DONJON ALBERTA=//

alberta,100,100,4	script	Explorateur#alb	4_F_NURSE,{
set @npc$, "[Explorateur]";

	if (questprogress(00000) == -1){
		mes @npc$;
		mes "Bonjour jeune aventurier, je peux t'emmener dans certains endroits reculé de cette ville,";
		mes "certains endroits où les plus gros monstres des environs rodes !";
		next;
		mes @npc$;
		mes "Que voulez-vous ?";
		switch(select("Parlez moi de votre société","Je souhaites me déplacer")){
			case 1:
				mes @npc$;
				mes "Grâce aux explorateurs de notre monde, nous avons créé Explorateur Corporate.";
				mes "Nous pouvons vous emmenez dans différents donjons aux alentours de certaines villes.";
				mes "Il y a peu nous avions un chef mais celui-ci est tombé sous les assaults des Orcs ...";
				next;
				mes @npc$;
				mes "Notre chef, Elriche était un très bon explorateur, il allait souvent avec sa femme découvrir les donjons de ce monde,";
				mes "mais un beau jour, après une exploration à l'Ouest de Geffen, accompagné de sa femme et d'un autre explorateur,";
				mes "il succomba...";
				next;
				mes @npc$;
				mes "Un seul survivra, il aurai réussi à s'échapper à temps pour éviter une mort certaine !";
				mes "Malheureusement il n'a pu sauvé Elriche et sa femme ...";
				mes "Depuis ce jour, le survivant qui se nomme Ariol est maintenant devenu notre chef !";
				next;
				mes @npc$;
				mes "Je vais m'arrêter là car on dit que le fait d'en parler continu de faire souffrir Ariol !";
				mes "Si tu veux vraiment en savoir plus, va donc le trouver, il se trouve à la taverne de la Cité Portuaire de Midgard.";
				setquest 00000;
				dispbottom "Vous avez reçu une nouvelle quête, vous pouvez la consulter via la commande alt+u";
				close;
			case 2:
				mes @npc$;
				mes "Voici les donjons accessible :";
				switch(select("Trutle Island","Sunken Ship")){
					case 1:
						mes @npc$;
						mes "Laisse moi le temps de me préparer pour l'expédition.";
						set Zeny,Zeny-5000;
						addtimer 10000, strnpcinfo(3)+"::OnWarp";
						set TURTLE, 1;
						set alreadytalk, 1;
						close;
					case 2:
						mes @npc$;
						mes "Laisse moi le temps de me préparer pour l'expédition.";
						set Zeny,Zeny-5000;
						addtimer 10000, strnpcinfo(3)+"::OnWarp";
						set sunken, 1;
						set alreadytalk, 1;
						close;
				}
		}
	}
	else if (questprogress(00000) == 1){
		mes @npc$;
		mes "Que voulez-vous ?";
		switch(select("Je souhaites me déplacer","Où se trouve Ariol ?")){
			case 1:
				mes @npc$;
				mes "Voici les donjons accessible :";
				switch(select("Trutle Island","Sunken Ship")){
					case 1:
						mes @npc$;
						mes "Laisse moi le temps de me préparer pour l'expédition.";
						set Zeny,Zeny-5000;
						addtimer 10000, strnpcinfo(3)+"::OnWarp";
						set TURTLE, 1;
						set alreadytalk, 1;
						close;
					case 2:
						mes @npc$;
						mes "Laisse moi le temps de me préparer pour l'expédition.";
						set Zeny,Zeny-5000;
						addtimer 10000, strnpcinfo(3)+"::OnWarp";
						set sunken, 1;
						set alreadytalk, 1;
						close;
				}
			case 2:
				mes @npc$;
				mes "Va donc trouver Ariol !";
				mes "Il se trouve à la taverne de la Cité Portuaire de Midgard.";
				close;
		}
	}
	else if (questprogress(00000) == 2){
		mes @npc$;
		mes "Oh je vois que tu as fini par rejoindre l'entreprise Explorateur Corporate finalement !";
		mes "Peu de gens y arrive ! Félicitations !";
		mes "Je te donne donc une réduction sur les tarifs !";
		next;
		mes @npc$;
		mes "Voici les donjons accessible :";
		switch(select("Trutle Island","Sunken Ship")){
			case 1:
				mes @npc$;
				mes "Laisse moi le temps de me préparer pour l'expédition.";
				set Zeny,Zeny-2500;
				addtimer 10000, strnpcinfo(3)+"::OnWarp";
				set TURTLE, 1;
				set alreadytalk, 1;
				close;
			case 2:
				mes @npc$;
				mes "Laisse moi le temps de me préparer pour l'expédition.";
				set Zeny,Zeny-2500;
				addtimer 10000, strnpcinfo(3)+"::OnWarp";
				set sunken, 1;
				set alreadytalk, 1;
				close;
		}
	}
	
	OnWarp:
		dispbottom "Vous allez être warp";
		if (TURTLE == 1){
			set alreadytalk, 0;
			warp "tur_dun01",0,0;
		}
		else if (sunken == 1){
			set alreadytalk, 0;
			warp "sunken",0,0;
		}
}
